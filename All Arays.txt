using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class HealthWall : MonoBehaviour
{
    public int currentHealth;
    public int maxhealth;
    public GameObject[] destroyWall;

	void Start ()   
    {
        currentHealth = maxhealth;
	}
	

	void Update ()
    {
		
	}
    public void LoseHealth(int damge)
    {
        currentHealth -= damge;
        if (currentHealth <= 0)
        {
            Destroy(gameObject);
            for (int i = 0; i < destroyWall.Length; i++)
            {
                //destroyWall[i].GetComponent<Rigidbody>().isKinematic = false;
            }
        }
    }
}
